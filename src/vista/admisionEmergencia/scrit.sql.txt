SELECT ACTO_MEDICO 
FROM CAJA_PREVENTA,ADMISION_EMERGENCIA_TRIAJE,ADMISION_EMERGENCIA_TOPICO
WHERE ADMISION_EMERGENCIA_TOPICO.ID_PREVENTA = CAJA_PREVENTA.Id_Preventa
AND ADMISION_EMERGENCIA_TRIAJE.ID_PREVENTA = CAJA_PREVENTA.Id_Preventa
AND ADMISION_EMERGENCIA_TRIAJE.TRIAJE_ID = ADMISION_EMERGENCIA_TOPICO.TRIAJE_ID
AND ADMISION_EMERGENCIA_TOPICO.ID_PREVENTA = 23

SELECT CAJA_JERARQUIA_FORMA_PAGO.NOM_FORMA_PAGO
FROM CAJA_DOCUMENTO_CABECERA,CAJA_JERARQUIA_FORMA_PAGO 
WHERE CAJA_DOCUMENTO_CABECERA.cod_jerar_forma_pago = CAJA_JERARQUIA_FORMA_PAGO.cod_jerar_forma_pago
AND Id_ActoMedico = 1149
